# Debugging a resource

You can use `useDebug` to log a handle in the console when it's updated.

```tsx
XSWR.useDebug(handle, label)
```

### Example

```tsx
function useMyData() {
  const handle = XSWR.useSingle<MyData>(
    `/api/hello`, 
    fetchAsJson)
  XSWR.useFetch(handle)
  XSWR.useDebug(handle, "my data")
  return handle
}
```

### Implementation

```tsx
export function useDebug<D = any, E = any>(
  handle: Handle<D, E>,
  label: string
) {
  const { time } = handle

  useEffect(() => {
    console.debug(label, handle)
  }, [time])
}
```
