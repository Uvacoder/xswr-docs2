# Debugging a resource

You can use `useDebug` to log a handle in the console when it's updated.

```tsx
XSWR.useDebug(handle, label)
```

### Example

```tsx
function useFetchAndDebugMixture(handle: XSWR.Handle, label: string) {
  XSWR.useFetch(handle
  XSWR.useDebug(handle, label)
  return handle
}

function useHelloMix() {
  const handle = XSWR.use(getHelloSchema, [])
  return useFetchAndDebugMixture(handle, label)
}
```

### Implementation

```tsx
export function useDebug<D = any, E = any>(
  handle: Handle<D, E>,
  label: string
) {
  const { time } = handle

  useEffect(() => {
    console.debug(label, handle)
  }, [time])
}
```
