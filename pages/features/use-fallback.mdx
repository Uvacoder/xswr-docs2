# Using SSR and ISR

You can use `useFallback` to display static or server-side data before the request is finished.

```tsx
XSWR.useFallback(handle, state) 
```

### Example

When using a Next.js like component

```tsx
function useFetchAndFallbackMixture<T>(handle: XSWR.Handle<T>, fallback?: T) {
  XSWR.useFetch(handle)
  XSWR.useFallback(handle, { data: fallback })
  return handle
}

function useHelloMix(fallback?: MyData) {
  const handle = XSWR.use(getHelloSchema, [])
  return useFetchAndFallbackMixture(handle, fallback)
}

function MyPage(props: { fallback?: Hello }) {
  const { data, error } = useHelloMix(props.fallback)

  return <>...</>
}
```

### Implementation

```tsx
export function useFallback<D = any, E = any>(
  handle: Handle<D, E>,
  state?: State<D, E>
) {
  const { data, error } = handle

  if (data !== undefined) return
  if (error !== undefined) return
  Object.assign(handle, state)
}
```
