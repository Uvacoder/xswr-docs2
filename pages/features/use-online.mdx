# Fetch when the browser is online

You can use `useOnline` to fetch a resource when the browser becomes online.

```tsx
XSWR.useOnline(handle) 
```

### Example

```tsx
function useMyData() {
  const handle = XSWR.useSingle<MyData>(
    `/api/hello`, 
    fetchAsJson)
  XSWR.useFetch(handle)
  XSWR.useOnline(handle)
  return handle
}
```

### Implementation

```tsx
export function useOnline(handle: Handle) {
  const { fetch } = handle

  useEffect(() => {
    const f = () => fetch()
    addEventListener("online", f)
    return () => removeEventListener("online", f)
  }, [fetch])
}
```
