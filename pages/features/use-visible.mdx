# Fetch when the tab is focused

You can use `useVisible` to fetch a resource when the tab is focused.

```typescript
XSWR.useVisible(handle) 
```

### Example

```typescript
function useMyData() {
  const handle = XSWR.useSingle<MyData>(
    `/api/hello`, 
    fetchAsJson)
  XSWR.useFetch(handle)
  XSWR.useVisible(handle)
  return handle
}
```

### Implementation

```typescript
export function useVisible(handle: Handle) {
  const { fetch } = handle

  useEffect(() => {
    const f = () => !document.hidden && fetch()
    document.addEventListener("visibilitychange", f)
    return () => document.removeEventListener("visibilitychange", f)
  }, [fetch])
}
```
